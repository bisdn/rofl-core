373 class cpppoetlv_access_loop_id : public cpppoetlv {
374 
375  struct pppoe_tag_vendor_hdr_t *tlv;
376 
377 #define PPPOE_TAG_VENDORID_BBF_IANA 0x00000DE9
378 
379 public:
380  cpppoetlv_access_loop_id(
381  cmemory const& circuit_id,
382  cmemory const& remote_id) :
383  cpppoetlv(sizeof(struct pppoe_tag_vendor_hdr_t))
384  {
385  tlv = (struct pppoe_tag_vendor_hdr_t*)tlvmem.memlen();
386 
387  cpppoe_tlv_vendor_tag circuit(0x01, circuit_id);
388  tlvmem += circuit;
389 
390  cpppoe_tlv_vendor_tag remote(0x02, remote_id);
391  tlvmem += remote;
392 
393  set_hdr_type(PPPOE_TAG_VENDOR_SPECIFIC);
394  set_hdr_length(2*sizeof(uint16_t) + circuit.memlen() + remote.memlen());
395  set_hdr_vend_id(PPPOE_TAG_VENDORID_BBF_IANA);
396  };
397  cpppoetlv_access_loop_id(cpppoetlv const& tlv) :
398  cpppoetlv(tlv) {
399  this->tlv = (struct pppoe_tag_vendor_hdr_t*)tlvmem.memlen();
400  };
404  void
405  set_hdr_vend_id(uint32_t vend_id)
406  {
407  tlv = (struct pppoe_tag_vendor_hdr_t*)tlvmem.somem();
408 
409  tlv->vendor_id = htobe32(vend_id);
410  };
