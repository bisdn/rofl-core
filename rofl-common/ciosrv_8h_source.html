280  cioloop(pthread_t tid = 0) :
281  tid(tid),
282  keep_on_running(false) {
283 
284  if (0 == tid) {
285  this->tid = pthread_self();
286  }
287 
288  struct rlimit rlim;
289  if (getrlimit(RLIMIT_NOFILE, &rlim) < 0) {
290  throw eSysCall("getrlimit()");
291  }
292  RwLock rlock(rfds_rwlock, RwLock::RWLOCK_WRITE);
293  RwLock wlock(wfds_rwlock, RwLock::RWLOCK_WRITE);
294  for (unsigned int i = 0; i < rlim.rlim_cur; i++) {
295  rfds.push_back(NULL);
296  wfds.push_back(NULL);
297  }
298  minrfd = rfds.size();
299  maxrfd = 0;
300  minwfd = wfds.size();
301  maxwfd = 0;
